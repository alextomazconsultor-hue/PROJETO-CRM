Para usar dados reais, crie um projeto em supabase.com, execute o SQL abaixo para criar as tabelas, e insira suas credenciais aqui.
Ver SQL para criar tabelas 📋 
CREATE TABLE leads (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  nome text, telefone text, email text,
  origem text, empreendimento text, cidade text,
  observacoes text, corretor text, status text,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE tarefas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  titulo text, lead_rel text, data_hora timestamptz,
  prioridade text, notas text, concluida boolean DEFAULT false,
  corretor text,
  created_at timestamptz DEFAULT now()
);
-- Se a tabela tarefas já existir sem a coluna corretor, rode também:
-- ALTER TABLE tarefas ADD COLUMN IF NOT EXISTS corretor text;
CREATE TABLE corretores (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  nome text UNIQUE,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE empreendimentos (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  nome text UNIQUE,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE historico (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lead_id uuid, texto text, tipo text,
  created_at timestamptz DEFAULT now()
);
-- Habilitar Row Level Security e criar políticas conforme necessidade
